#include <iostream>
using namespace std;

inline double F(int k, int b, int x) {
	return k * x + b;
}

int main() {
	int x[3], y[3], t = 1;					//1-小在上，斜率正，中在右(逆)
	for (int i = 0; i < 3; i++)
		cin >> x[i] >> y[i];
	int high, mid, low;
	high = y[0] >= y[1] ? (y[0] >= y[2] ? y[0] : y[2]) : (y[1] >= y[2] ? y[1] : y[2]);
	//若出现相等，得到的high是较小的下标
	low = y[0] < y[1] ? (y[0] < y[2] ? y[0] : y[2]) : (y[1] < y[2] ? y[1] : y[2]);
	mid = 3 - high - low;
	if (high == mid) {
		if (x[high] > x[mid]) t = -t;
		if (high < low)t = -t;
	}
	else if (x[high] != x[low]) {
		double k = (y[high] - y[low]) / (x[high] - x[low]);
		double b = y[0] - k * x[0];
		if (high > low)t = -t;
		if (k < 0) t = -t;
		//if (y[0] > y[1]) t = -t;
		if (y[mid] > F(k, b, x[mid])) t = -t;
	}
	else {
		if (high > low) t = -t;
		if (x[mid] < x[high]) t = -t;
	}
	if (t == 1) cout << 0;
	else cout << 1;

	return 0;
}
